# -----------------------------------------------------------------------------
# CORE FRAMEWORK
# -----------------------------------------------------------------------------

# Flask - Web framework for building the backend API
# Used to create REST endpoints for problem solving and quiz generation
flask>=3.0.0

# Flask-CORS - Cross-Origin Resource Sharing support for Flask
# Required to allow the frontend to communicate with the backend
flask-cors>=4.0.0

# Python-dotenv - Load environment variables from .env file
# Used to load Google Client ID and other configuration
python-dotenv>=1.0.0

# -----------------------------------------------------------------------------
# AI/LLM INTEGRATION
# -----------------------------------------------------------------------------

# Google Generative AI - Official Python SDK for Gemini API (FREE!)
# Used to send requests to Gemini for solving math problems and generating quizzes
# Each user provides their own API key
google-generativeai>=0.3.0

# -----------------------------------------------------------------------------
# FILE UPLOAD & PROCESSING
# -----------------------------------------------------------------------------

# Pillow - Python Imaging Library
# Required for processing uploaded image files (PNG, JPG, etc.)
# Used to convert and prepare images for the Gemini Vision API
Pillow>=10.0.0

# PyMuPDF (fitz) - PDF processing library
# Used to extract text and convert PDF pages to images
# This is the PRIMARY PDF processing library for this application
PyMuPDF>=1.23.0

# pdf2image - Convert PDF pages to images (FALLBACK)
# Alternative method for PDF processing if PyMuPDF is unavailable
# Requires poppler-utils to be installed on the system
pdf2image>=1.16.0

# python-docx - Microsoft Word document processing
# Used to extract text from DOCX files uploaded by users
python-docx>=1.1.0

# =============================================================================
# INSTALLATION INSTRUCTIONS
# =============================================================================
#
# 1. Create a virtual environment (recommended):
#    python -m venv venv
#    source venv/bin/activate  # On Windows: venv\Scripts\activate
#
# 2. Install dependencies:
#    pip install -r requirements.txt
#
# 3. For PDF processing, also install system dependencies:
#    Ubuntu/Debian: sudo apt-get install poppler-utils
#    macOS: brew install poppler
#
# 4. Run the server:
#    python server.py
#
